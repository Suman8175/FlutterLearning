To give design to button use 
 style: const ButtonStyle(
                    backgroundColor: MaterialStatePropertyAll(Colors.redAccent),
                  )
in ElevatedButton Widjet or Any Button Widjet


Example 1:

 

import 'dart:developer';



import 'package:flutter/material.dart';



void main() {

  runApp(

    MaterialApp(

      home: Scaffold(

        appBar: AppBar(),

        body: ElevatedButton(

          child: Text('Button'),

          onPressed: () {

            print('Press');

          },

          onLongPress: () {

            print('Long Press');

          },

        ),

      ),

    ),

  );

}
